&= \mathbf{I} - \sum{t=1}^{n-1} \mathbf{w}_t \mathbf{k}_t^\top - \beta_n \mathbf{k}_n \mathbf{k}n^\top + \left(\sum{t=1}^{n-1} \mathbf{w}_t \mathbf{k}_t^\top\right) \beta_n \mathbf{k}_n \mathbf{k}n^\top \
&= \mathbf{I} - \sum{t=1}^{n-1} \mathbf{w}_t \mathbf{k}_t^\top - \underbrace{\left(\beta_n \mathbf{k}n - \beta_n \sum{t=1}^{n-1} \left(\mathbf{w}_t (\mathbf{k}_t^\top \mathbf{k}n)\right)\right)}{\mathbf{w}_n} \mathbf{k}n^\top \
&= \mathbf{I} - \sum{t=1}^{n} \mathbf{w}_t \mathbf{k}_t^\top
\end{aligned}
